def solve_n_queens(n):
    board = [['.' for _ in range(n)] for _ in range(n)]

    cols = set()
    diag1 = set()   # row - col
    diag2 = set()   # row + col

    def solve(row):
        if row == n:
            for r in board:
                print(" ".join(r))
            return True

        for col in range(n):

            # Branch and Bound pruning
            if col in cols or (row - col) in diag1 or (row + col) in diag2:
                continue

            # Place queen
            board[row][col] = 'Q'
            cols.add(col)
            diag1.add(row - col)
            diag2.add(row + col)

            # Backtracking
            if solve(row + 1):
                return True

            # Remove queen
            board[row][col] = '.'
            cols.remove(col)
            diag1.remove(row - col)
            diag2.remove(row + col)

        return False

    solve(0)


# Main
n = 4
solve_n_queens(n)
